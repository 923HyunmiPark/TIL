## 1번
scores = [80, 100, 70, 90, 40]
def total(scores):
    sum = 0
    for i in range(0, len(scores)):
        sum += scores[i]
    return(sum)
print (total(scores))

scores = [80, 100, 70, 90, 40]
def total(scores):
    sum = 0
    for i in range(0, len(scores)):
        sum += scores[i]
        average = sum/len(scores)
    return(average)
print (total(scores))
